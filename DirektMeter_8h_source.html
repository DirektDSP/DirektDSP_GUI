158 {
159 float targetDb = config.rangeMinDb;
160 if (source != nullptr)
161 targetDb = source->load (std::memory_order_relaxed);
162
163 targetDb = juce::jlimit (config.rangeMinDb, config.rangeMaxDb, targetDb);
164
165 if (targetDb >= 0.0f)
166 clipped = true;
167
168 // Mode-specific smoothing coefficients.
169 // Rms/Lufs use slower ballistics to approximate integration time.
170 float attackCoeff = 0.6f;
171 float releaseCoeff = 0.15f;
172 if (config.mode == Mode::Rms || config.mode == Mode::Lufs)
173 {
174 attackCoeff = 0.2f;
175 releaseCoeff = 0.05f;
176 }
177
178 float coeff = (targetDb > smoothedDb) ? attackCoeff : releaseCoeff;
179 smoothedDb += coeff * (targetDb - smoothedDb);
180
181 // Peak hold
182 if (targetDb > peakDb)
183 {
184 peakDb = targetDb;
185 peakHoldCounter = config.refreshHz; // ~1 second hold
186 }
187 else if (peakHoldCounter > 0)
188 {
189 --peakHoldCounter;
190 }
191 else
192 {
193 peakDb += releaseCoeff * (config.rangeMinDb - peakDb);
194 }
195
196 repaint();
197 }
