20void DirektLookAndFeel::drawRotarySlider (juce::Graphics& g, int x, int y, int width, int height, float sliderPos,
21 float rotaryStartAngle, float rotaryEndAngle, juce::Slider& slider)
22{
23 juce::ignoreUnused (slider);
24
25 auto bounds = juce::Rectangle<int> (x, y, width, height).toFloat().reduced (4.0F);
26 auto radius = juce::jmin (bounds.getWidth(), bounds.getHeight()) / 2.0F;
27 auto centreX = bounds.getCentreX();
28 auto centreY = bounds.getCentreY();
29 auto rx = centreX - radius;
30 auto ry = centreY - radius;
31 auto rw = radius * 2.0F;
32 auto toAngle = rotaryStartAngle + sliderPos * (rotaryEndAngle - rotaryStartAngle);
33 auto lineW = juce::jmax (2.0F, radius * 0.12F);
34 auto arcRadius = radius - lineW * 0.5F;
35
36 // Background circle
37 g.setColour (Colours::knobBg);
38 g.fillEllipse (rx, ry, rw, rw);
39
40 // Track arc (full range)
41 juce::Path bgArc;
42 bgArc.addCentredArc (centreX, centreY, arcRadius, arcRadius, 0.0F, rotaryStartAngle, rotaryEndAngle, true);
43 g.setColour (Colours::knobTrack);
44 g.strokePath (bgArc, juce::PathStrokeType (lineW, juce::PathStrokeType::curved, juce::PathStrokeType::rounded));
45
46 // Value arc
47 if (sliderPos > 0.0F)
48 {
49 juce::Path valueArc;
50 valueArc.addCentredArc (centreX, centreY, arcRadius, arcRadius, 0.0F, rotaryStartAngle, toAngle, true);
51 g.setColour (accentColour);
52 g.strokePath (valueArc,
53 juce::PathStrokeType (lineW, juce::PathStrokeType::curved, juce::PathStrokeType::rounded));
54 }
55
56 // Glow behind value arc
57 if (sliderPos > 0.0F)
58 {
59 auto glowColour = accentColour.withAlpha (0.15F);
60 juce::Path glowArc;
61 glowArc.addCentredArc (centreX, centreY, arcRadius, arcRadius, 0.0F, rotaryStartAngle, toAngle, true);
62 g.setColour (glowColour);
63 g.strokePath (glowArc,
64 juce::PathStrokeType (lineW * 3.0F, juce::PathStrokeType::curved, juce::PathStrokeType::rounded));
65 }
66
67 // Pointer line
68 juce::Path pointer;
69 auto pointerLength = radius * 0.6F;
70 auto pointerThickness = juce::jmax (1.5F, radius * 0.08F);
71 pointer.addRoundedRectangle (-pointerThickness * 0.5F, -pointerLength, pointerThickness, pointerLength, 1.0F);
72 pointer.applyTransform (juce::AffineTransform::rotation (toAngle).translated (centreX, centreY));
73 g.setColour (Colours::knobPointer);
74 g.fillPath (pointer);
75
76 // Center dot
77 auto dotRadius = radius * 0.12F;
78 g.setColour (Colours::knobPointer);
79 g.fillEllipse (centreX - dotRadius, centreY - dotRadius, dotRadius * 2.0F, dotRadius * 2.0F);
80}

82void DirektLookAndFeel::drawToggleButton (juce::Graphics& g, juce::ToggleButton& button,
83 bool shouldDrawButtonAsHighlighted, bool /*shouldDrawButtonAsDown*/)
84{

89 indicatorSize, indicatorSize);
90
91 // Indicator background

94
95 // Indicator border
96 g.setColour (shouldDrawButtonAsHighlighted ? Colours::textDim : Colours::divider);
97 g.drawRoundedRectangle (indicatorBounds, 3.0F, 1.0F);
98
99 if (button.getToggleState())
100 {
101 auto inner = indicatorBounds.reduced (3.0F);
102 g.setColour (accentColour);
103 g.fillRoundedRectangle (inner, 2.0F);
104
105 // Glow
106 g.setColour (accentColour.withAlpha (0.2F));
107 g.fillRoundedRectangle (indicatorBounds.expanded (2.0F), 4.0F);
108 }
109
110 // Label text
111 auto textBounds = bounds.withTrimmedLeft (indicatorSize + 8.0F);
112 g.setColour (Colours::textLabel);
113 g.setFont (juce::Font (juce::FontOptions (juce::jmin (h * 0.55F, 13.0F))));
114 g.drawText (button.getButtonText(), textBounds.toNearestInt(), juce::Justification::centredLeft);
