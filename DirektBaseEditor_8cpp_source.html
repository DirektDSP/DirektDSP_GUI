202bool DirektBaseEditor::keyPressed (const juce::KeyPress& key)
203{
204 auto const isCommandDown = key.getModifiers().isCommandDown();
205 if (!isCommandDown)
206 {
207 return AudioProcessorEditor::keyPressed (key);
208 }
209
210 auto const keyCode = key.getKeyCode();
211 if (keyCode == 'z' || keyCode == 'Z')
212 {
213 auto const wantsRedo = key.getModifiers().isShiftDown();
214 return wantsRedo ? parameterHistory.redo (apvts) : parameterHistory.undo (apvts);
215 }
216
217 if (keyCode == 'y' || keyCode == 'Y')
218 {
219 return parameterHistory.redo (apvts);
220 }
221
222 return AudioProcessorEditor::keyPressed (key);

274juce::Component* DirektBaseEditor::findComponentByID (const juce::String& id) const
275{
276 // Search recursively through all children
277 std::function<juce::Component*(juce::Component*)> search;
278 search = [&] (juce::Component* parent) -> juce::Component*
279 {
280 if (parent->getComponentID() == id)
281 {
282 return parent;
283 }
284 for (auto* child : parent->getChildren())
285 {
286 if (auto* found = search (child))
287 {
288 return found;
289 }
290 }
291 return nullptr;
292 };
293
294 return search (const_cast<DirektBaseEditor*> (this));
295}

297void DirektBaseEditor::bindMeterSource (const juce::String& sourceID, const std::atomic<float>* source)
298{
299 if (source == nullptr)
300 {
301 return;
302 }
303
304 buildContext.meterSources[sourceID] = source;
305
306 // For meters/indicators that need post-build connection, search by component ID
307 if (auto* comp = findComponentByID (sourceID))
308 {
309 if (auto* meter = dynamic_cast<DirektMeter*> (comp))
310 {
311 meter->setSource (source);
312 }
313 else if (auto* indicator = dynamic_cast<DirektClipIndicator*> (comp))
314 {
315 indicator->setSource (source);
316 }
317 }
318
319 // Walk the tree to bind DirektStereoMeter and DirektClipIndicator by source ID,
320 // which allows the component ID and source ID to differ.
321 std::function<void (juce::Component*)> bindBySourceID;
322 bindBySourceID = [&] (juce::Component* parent)
323 {
324 if (auto* stereoMeter = dynamic_cast<DirektStereoMeter*> (parent))
325 {
326 stereoMeter->tryBindSource (sourceID, source);
327 }
328 else if (auto* indicator = dynamic_cast<DirektClipIndicator*> (parent))
329 {
330 indicator->tryBindSource (sourceID, source);
331 }
332 for (auto* child : parent->getChildren())
333 {
334 bindBySourceID (child);
335 }
336 };
337 bindBySourceID (this);
338}

340void DirektBaseEditor::configureCrashRecovery (const juce::String& pluginName)
341{
342 if (!juce::JUCEApplicationBase::isStandaloneApp())
343 {
344 return;
345 }
346
347 auto safePluginName = pluginName.retainCharacters (safeFilenameChars);
348 if (safePluginName.isEmpty())
349 {
350 safePluginName = "DirektDSP";
351 }
352
353 const auto autosaveDirectory = juce::File::getSpecialLocation (juce::File::userApplicationDataDirectory)
354 .getChildFile ("DirektDSP")
355 .getChildFile ("crash-recovery");
356
357 if (!autosaveDirectory.exists() && !autosaveDirectory.createDirectory())
358 {
359 return;
360 }
361
362 crashRecoveryFile = autosaveDirectory.getChildFile (safePluginName + "-autosave.xml");
363 crashRecoveryEnabled = true;
364
365 if (crashRecoveryFile.existsAsFile())
366 {
367 auto safeThis = juce::Component::SafePointer<DirektBaseEditor> (this);
368 juce::MessageManager::callAsync (
369 [safeThisCopy = safeThis]
370 {
371 if (safeThisCopy != nullptr)
372 {
373 safeThisCopy->restoreCrashRecoveryIfNeeded();
374 }
375 });
376 }
377
378 saveCrashRecoverySnapshot();
379 startTimer (autosaveIntervalMs);
380}

404void DirektBaseEditor::restoreCrashRecoveryIfNeeded()
405{
406 if (!crashRecoveryFile.existsAsFile())
407 {
408 return;
409 }
410
411 const auto shouldRestore = juce::AlertWindow::showOkCancelBox (
412 juce::MessageBoxIconType::QuestionIcon, "Restore previous session?",
413 "DirektDSP found autosaved settings from an unexpected shutdown.\n\nRestore these settings?", "Restore",
414 "Discard", this, nullptr);
415
416 if (shouldRestore)
417 {
418 auto restored = false;
419
420 if (auto xml = juce::parseXML (crashRecoveryFile))
421 {
422 auto restoredState = juce::ValueTree::fromXml (*xml);
423 if (restoredState.isValid())
424 {
425 apvts.replaceState (restoredState);
426 header.updatePresetName();
427 restored = true;
428 }
429 }
430
431 if (!restored)
432 {
433 crashRecoveryFile.deleteFile();
434 juce::AlertWindow::showMessageBoxAsync (juce::MessageBoxIconType::WarningIcon,
435 "Autosave could not be restored",
436 "The autosave data appears to be invalid and has been discarded.",
437 {}, this);
438 }
439 }
440 else
441 {
442 crashRecoveryFile.deleteFile();
443 }
444
445 saveCrashRecoverySnapshot();
446}
