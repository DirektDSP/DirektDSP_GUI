233juce::Component* DirektBaseEditor::findComponentByID (const juce::String& id) const
234{
235 // Search recursively through all children
236 std::function<juce::Component*(juce::Component*)> search;
237 search = [&] (juce::Component* parent) -> juce::Component*
238 {
239 if (parent->getComponentID() == id)
240 {
241 return parent;
242 }
243 for (auto* child : parent->getChildren())
244 {
245 if (auto* found = search (child))
246 {
247 return found;
248 }
249 }
250 return nullptr;
251 };
252
253 return search (const_cast<DirektBaseEditor*> (this));
254}

256void DirektBaseEditor::bindMeterSource (const juce::String& sourceID, const std::atomic<float>* source)
257{
258 buildContext.meterSources[sourceID] = source;
259
260 // If the tree is already built, find any meters with matching sourceID and connect them
261 std::function<void (juce::Component*)> connectMeters;
262 connectMeters = [&] (juce::Component* parent)
263 {
264 if (auto* meter = dynamic_cast<DirektMeter*> (parent))
265 {
266 // Meters don't expose their sourceID, but we can use component ID
267 // In practice, meters should be rebuilt or we store the mapping
268 meter->setSource (source);
269 }
270 for (auto* child : parent->getChildren())
271 {
272 if (child != nullptr)
273 {
274 connectMeters (child);
275 }
276 }
277 };
278
279 // For meters that need post-build connection, search by component ID
280 if (auto* comp = findComponentByID (sourceID))
281 {
282 if (auto* meter = dynamic_cast<DirektMeter*> (comp))
283 {
284 meter->setSource (source);
285 }
286 }
287}
