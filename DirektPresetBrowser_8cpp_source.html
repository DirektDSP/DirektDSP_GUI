181void DirektPresetBrowser::refreshTags()
182{
183 categoryModel.categories.clear();
184 categoryModel.categories.add ("All");
185 categoryModel.categories.add ("Favorites");
186
187 auto metadata = presetManager.getAllPresetMetadata();
188 juce::StringArray tags;
189
190 for (auto const& item : metadata)
191 {
192 if (item.category.isNotEmpty())
193 {
194 tags.addIfNotAlreadyThere ("#" + item.category);
195 }
196 if (item.artist.isNotEmpty())

201 tags.sort (true);

211void DirektPresetBrowser::refreshPresets()
212{
213 presetModel.presets.clear();
214
215 juce::String selectedCategoryOrTag;
216 int const catRow = categoryList.getSelectedRow();
217 if (catRow > 0 && catRow < categoryModel.categories.size())
218 {
219 selectedCategoryOrTag = categoryModel.categories[catRow];
220 }
221
222 auto metadata = presetManager.getAllPresetMetadata();
223 auto const searchText = searchEditor.getText().toLowerCase().trim();
224 auto const useSearch = searchText.isNotEmpty();
225
226 auto tagPasses = [this, &selectedCategoryOrTag] (const Service::PresetMetadata& item)
227 {
228 if (selectedCategoryOrTag.isEmpty() || selectedCategoryOrTag == "All")
229 {
230 return true;
231 }
232 if (selectedCategoryOrTag == "Favorites")
233 {
234 auto const key = makeFavoriteKey (item.name, item.category).toStdString();
235 return favoritePresetKeys.find (key) != favoritePresetKeys.end();
236 }
237 if (selectedCategoryOrTag.startsWithChar ('#'))
238 return item.category == selectedCategoryOrTag.substring (1);
239 if (selectedCategoryOrTag.startsWithChar ('@'))
240 return item.artist == selectedCategoryOrTag.substring (1);
241 return false;
242 };
243
244 auto searchPasses = [&searchText, useSearch] (const Service::PresetMetadata& item)
245 {
246 if (!useSearch)
247 return true;
248 auto haystack =
249 (item.name + " " + item.artist + " " + item.category + " " + item.tags.joinIntoString (" ")).toLowerCase();
250 return haystack.contains (searchText);
251 };
252
253 for (auto const& item : metadata)
254 {
255 if (!tagPasses (item) || !searchPasses (item))
256 continue;
257
258 presetModel.presets.push_back ({item.name, item.artist, item.category});
259 }
260
261 std::sort (presetModel.presets.begin(), presetModel.presets.end(),
262 [] (const PresetEntry& lhs, const PresetEntry& rhs)
263 { return lhs.name.compareIgnoreCase (rhs.name) < 0; });
264
265 presetModel.selectedRow = -1;
266 presetList.updateContent();
267 presetList.deselectAllRows();
268 presetList.repaint();
269}

487void DirektPresetBrowser::loadFavorites()
488{
489 favoritePresetKeys.clear();
490
491 auto favoritesFile = juce::File::getSpecialLocation (juce::File::userApplicationDataDirectory)
492 .getChildFile ("DirektDSP")
493 .getChildFile ("preset-browser-favorites.json");
494
495 if (!favoritesFile.existsAsFile())
496 {
497 return;
498 }
499
500 auto parsed = juce::JSON::parse (favoritesFile);
501 auto* list = parsed.getArray();
502 if (list == nullptr)
503 {
504 return;
505 }
506
507 for (auto const& value : *list)
508 {
509 auto key = value.toString().trim();
510 if (key.isNotEmpty())
511 {
512 favoritePresetKeys.insert (key.toStdString());
513 }
514 }
