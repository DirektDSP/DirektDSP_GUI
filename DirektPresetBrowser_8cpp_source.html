147void DirektPresetBrowser::refreshPresets()
148{
149 presetModel.presets.clear();
150
151 juce::String selectedCategory;
152 int const catRow = categoryList.getSelectedRow();
153 if (catRow > 0 && catRow < categoryModel.categories.size())
154 {
155 selectedCategory = categoryModel.categories[catRow];
156 }
157
158 if (selectedCategory.isEmpty())
159 {
160 // "All" — show everything
161 auto metadata = presetManager.getAllPresetMetadata();
162 for (auto& m : metadata)
163 {
164 presetModel.presets.push_back ({m.name, m.artist, m.category});
165 }
166 }
167 else
168 {
169 auto metadata = presetManager.getPresetMetadataInCategory (selectedCategory);
170 for (auto& m : metadata)
171 {
172 presetModel.presets.push_back ({m.name, m.artist, m.category});
173 }
174 }
175
176 presetModel.selectedRow = -1;
177 presetList.updateContent();
178 presetList.deselectAllRows();
179 presetList.repaint();
180}
