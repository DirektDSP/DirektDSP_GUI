261// Helpers for container nodes that recurse via the registry
262BuiltNode buildSectionNode (const SectionDesc& desc, BuildContext& ctx, const DirektComponentRegistry& reg)
263{
264 auto section = std::make_unique<DirektSection> (desc.title, desc.columns);
265 applyNodeProps (*section, desc.props);
266 if (auto strip =
267 DirektModuleBypassSoloStrip::tryCreate (ctx.apvts, desc.bypassParamID, desc.soloParamID, desc.bypassLabel,
268 desc.soloLabel, desc.bypassTooltip, desc.soloTooltip))
269 {
270 section->setTitleBarAccessory (std::move (strip));
271 }
272 std::vector<std::unique_ptr<juce::Component>> owned;
273 for (const auto& childNode : desc.children)
274 {
275 if (!childNode)
276 {
277 continue;
278 }
279 auto built = reg.build (*childNode, ctx);
280 section->addControl (built.component.get());
281 owned.push_back (std::move (built.component));
282 for (auto& oc : built.ownedChildren)
283 {
284 owned.push_back (std::move (oc));
285 }
286 }
287 return {std::move (section), std::move (owned)};
288}

290BuiltNode buildFlexNode (const std::vector<Node>& children, const NodeProps& props, BuildContext& ctx,
291 const DirektComponentRegistry& reg, DirektFlexContainer::Direction dir)
292{
293 auto container = std::make_unique<DirektFlexContainer> (dir);
294 applyNodeProps (*container, props);
295 std::vector<std::unique_ptr<juce::Component>> owned;
296 for (const auto& childNode : children)
297 {
298 if (!childNode)
299 {
300 continue;
301 }
302 auto childSize = std::visit ([] (const auto& d) { return d.props.size; }, *childNode);
303 auto built = reg.build (*childNode, ctx);
304 container->addFlexChild (built.component.get(), childSize);
305 owned.push_back (std::move (built.component));
306 for (auto& oc : built.ownedChildren)
307 {
308 owned.push_back (std::move (oc));
309 }
310 }
311
312 // Activate drag-to-reorder after all children have been added so that
313 // callbacks are wired to every DirektSection child in one pass.
314 if (props.draggable)
315 {
316 container->enableDragReorder (true);
317 }
318
319 return {std::move (container), std::move (owned)};
320}

370 BuiltNode operator() (const MacroDesc& d) const { return buildMacroNode (d, *ctx); }
371 BuiltNode operator() (const SliderDesc& d) const { return buildSliderNode (d, *ctx); }
372 BuiltNode operator() (const ButtonDesc& d) const { return buildButtonNode (d, *ctx); }
