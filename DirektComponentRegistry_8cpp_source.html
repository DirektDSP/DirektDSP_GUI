149BuiltNode buildMacroNode (const MacroDesc& desc, BuildContext& ctx)
150{
151 std::vector<MacroTarget> targets;
152 targets.reserve (desc.targets.size());
153
154 for (const auto& targetDesc : desc.targets)
155 {
156 if (auto* parameter = ctx.apvts.getParameter (targetDesc.paramID))
157 {
158 MacroTarget target;
159 target.parameter = parameter;
160 target.minNormalized = targetDesc.minNormalized;
161 target.maxNormalized = targetDesc.maxNormalized;
162 target.curveExponent = targetDesc.curveExponent;
163 targets.push_back (target);
164 }
165 }
166
167 auto macro = std::make_unique<DirektMacroControl> (desc.label, std::move (targets));
168 if (desc.tooltip.isNotEmpty())
169 {
170 macro->getSlider().setTooltip (desc.tooltip);
171 }
172 applyNodeProps (*macro, desc.props);
173 return {std::move (macro), {}};

209BuiltNode buildSectionNode (const SectionDesc& desc, BuildContext& ctx, const DirektComponentRegistry& reg)
210{
211 auto section = std::make_unique<DirektSection> (desc.title, desc.columns);
212 applyNodeProps (*section, desc.props);
213 if (auto strip =
214 DirektModuleBypassSoloStrip::tryCreate (ctx.apvts, desc.bypassParamID, desc.soloParamID, desc.bypassLabel,
215 desc.soloLabel, desc.bypassTooltip, desc.soloTooltip))
216 {
217 section->setTitleBarAccessory (std::move (strip));
218 }
219 std::vector<std::unique_ptr<juce::Component>> owned;
220 for (const auto& childNode : desc.children)
221 {
222 if (!childNode)
223 {
224 continue;
225 }
226 auto built = reg.build (*childNode, ctx);
227 section->addControl (built.component.get());
228 owned.push_back (std::move (built.component));
229 for (auto& oc : built.ownedChildren)
230 {
231 owned.push_back (std::move (oc));
232 }
233 }
234 return {std::move (section), std::move (owned)};

297BuiltNode DirektComponentRegistry::build (const NodeDescriptor& descriptor, BuildContext& ctx) const
298{
299 return std::visit (
300 [&] (const auto& desc) -> BuiltNode
301 {
302 using T = std::decay_t<decltype (desc)>;
303 if constexpr (std::is_same_v<T, KnobDesc>)
304 {
305 return buildKnobNode (desc, ctx);
306 }
307 else if constexpr (std::is_same_v<T, ToggleDesc>)
308 {
309 return buildToggleNode (desc, ctx);
310 }
311 else if constexpr (std::is_same_v<T, ComboBoxDesc>)
312 {
313 return buildComboBoxNode (desc, ctx);
314 else if constexpr (std::is_same_v<T, MacroKnobDesc>)
315 return buildMacroKnobNode (desc, ctx);
316 }
317 else if constexpr (std::is_same_v<T, SliderDesc>)
318 {
319 return buildSliderNode (desc, ctx);
320 }
321 else if constexpr (std::is_same_v<T, ButtonDesc>)
322 {
323 return buildButtonNode (desc, ctx);
324 }
325 else if constexpr (std::is_same_v<T, RadioGroupDesc>)
326 {
327 return buildRadioGroupNode (desc, ctx);
328 }
329 else if constexpr (std::is_same_v<T, XYPadDesc>)
330 {
331 auto comp = std::make_unique<juce::Component>();
332 applyNodeProps (*comp, desc.props);
333 return {std::move (comp), {}};
334 }
335 else if constexpr (std::is_same_v<T, MacroDesc>)
336 {
337 return buildMacroNode (desc, ctx);
338 }
339 else if constexpr (std::is_same_v<T, MeterDesc>)
340 {
341 return buildMeterNode (desc, ctx);
342 }
343 else if constexpr (std::is_same_v<T, LabelDesc>)
344 {
345 return buildLabelNode (desc, ctx);
346 }
347 else if constexpr (std::is_same_v<T, SpacerDesc>)
348 {
349 auto spacer = std::make_unique<juce::Component>();
350 applyNodeProps (*spacer, desc.props);
351 return {std::move (spacer), {}};
352 }
353 else if constexpr (std::is_same_v<T, DividerDesc>)
354 {
355 return buildDividerNode (desc, ctx);
356 }
357 else if constexpr (std::is_same_v<T, SectionDesc>)
358 {
359 return buildSectionNode (desc, ctx, *this);
360 }
361 else if constexpr (std::is_same_v<T, HBoxDesc>)
362 {
363 return buildFlexNode (desc.children, desc.props, ctx, *this, DirektFlexContainer::Direction::Row);
364 }
365 else if constexpr (std::is_same_v<T, VBoxDesc>)
366 {
367 return buildFlexNode (desc.children, desc.props, ctx, *this, DirektFlexContainer::Direction::Column);
368 }
369 else if constexpr (std::is_same_v<T, TabPanelDesc>)
370 {
371 return buildTabPanelNode (desc, ctx, *this);
372 }
373 else if constexpr (std::is_same_v<T, CustomDesc>)
374 {
375 return buildCustomNode (desc, ctx, customFactories);
376 }
377 else
378 {
379 auto comp = std::make_unique<juce::Component>();
380 return {std::move (comp), {}};
381 }
382 },
383 descriptor);
384}
